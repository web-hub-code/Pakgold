    <script>
        const answerKey = { q1:"A", q2:"B", q3:"A", q4:"A", q5:"A", q6:"A", q7:"B", q8:"A", q9:"A", q10:"A" };

        function calculateResult() {
            let marks = 0;
            const totalQuestions = Object.keys(answerKey).length;
            const formData = new FormData(document.getElementById('fullPaperForm'));

            for(let q in answerKey) {
                if(formData.get(q) === answerKey[q]) marks++;
            }

            const percentage = Math.round((marks / totalQuestions) * 100);
            document.getElementById('finalScore').innerText = percentage + "%";
            document.getElementById('feedback').innerText = percentage >= 50 ? "Qualified" : "Not Qualified";
            document.getElementById('resultModal').classList.remove('hidden');
        }

        let timeRemaining = 5400;
        setInterval(() => {
            if(timeRemaining <= 0) return;
            timeRemaining--;
            let mins = Math.floor(timeRemaining / 60);
            let secs = timeRemaining % 60;
            document.getElementById('timer').innerText = `${mins}:${secs < 10 ? '0' : ''}${secs}`;
        }, 1000);
    </script>
