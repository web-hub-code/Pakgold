    <script>
        const officialKey = { e1:"B", e2:"B", u1:"B", u2:"A", i1:"A", g1:"B", m1:"A" };

        function showFinalResult() {
            let score = 0;
            const total = Object.keys(officialKey).length;
            const form = new FormData(document.getElementById('officialExamForm'));

            for(let key in officialKey) {
                if(form.get(key) === officialKey[key]) score++;
            }

            const p = Math.round((score/total)*100);
            document.getElementById('scoreDisplay').innerText = p + "%";
            document.getElementById('statusFeedback').innerText = p >= 50 ? "Mubarak Ho! You are Qualified." : "Fail! Hard Work Needed.";
            document.getElementById('resultModal').classList.remove('hidden');
        }

        let totalTime = 5400;
        setInterval(() => {
            if(totalTime <= 0) return;
            totalTime--;
            let mm = Math.floor(totalTime/60);
            let ss = totalTime%60;
            document.getElementById('timer').innerText = `${mm}:${ss < 10 ? '0' : ''}${ss}`;
        }, 1000);
    </script>
